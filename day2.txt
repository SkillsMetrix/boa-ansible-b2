# importing package
from fastapi import FastAPI

app= FastAPI()
#dummy record
my_post=[{"id":"101","title":"Demo App","content":"demo content"}]
#decalare model
class Post(BaseModel):
    title:str
    content:str
#get request
@app.get("/showusers")
def showUsers():
    return {"message":my_post}
#post request
@app.post("/adduser")
def addUser(payload: Post):
    udata=payload.dict()

    my_post.append(udata)
    return{ "userdata":payload}
